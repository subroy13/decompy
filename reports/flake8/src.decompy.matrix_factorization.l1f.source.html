113 def _solve_ml1(self, D: np.ndarray, A: np.ndarray, pinvA: np.ndarray):
114 """Solve the L1-norm minimization problem.
115  
116 Parameters
117 ----------
118 D : ndarray
119 The n x s matrix D.
120 A : ndarray
121 The n x m matrix A.
122 pinvA : ndarray
123 The pseudo-inverse of A. If None, it will be computed.
124  
125 Returns
126 -------
127 E : ndarray
128 The n x s matrix E.
129 alpha : ndarray
130 The m x s matrix alpha.
131 niter : int
132 Number of iterations performed.
133 stop_c : float
134 The maximum absolute value of the residual.
135  
136 Solves the optimization problem:
137 min_{E,alpha} |E|_1
138 s.t A*alpha+E=D
139 Where E, D are n x s matrices,
140 A is a n x m matrix, and alpha is a m x s matrix.
141 """
142 n, m = A.shape
143 s = D.shape[1]
144  
145 if pinvA is None:
146 pinvA = np.linalg.pinv(A)
147  
148 # initialization
149 alpha = np.zeros((m, s))
150 E = np.zeros((n, s))
151 Y = np.zeros((n, s))
152 mu = 1e-6
153  
154 # start main loop
155 niter = 0
156 while niter < self.maxiter:
157 niter += 1
158  
159 # update E
160 temp_T = D - A @ alpha + (1 / mu) * Y # (n, s)
161 E = np.maximum(temp_T - 1 / mu, 0) + np.minimum(temp_T + 1 / mu, 0)
162  
163 # update alpha
164 temp_T = D + (1 / mu) * Y - E
165 alpha = pinvA @ temp_T
166  
167 leq = D - A @ alpha - E
168 stop_c = (np.abs(leq)).max()
169  
170 if stop_c < self.tol:
171 break
172 else:
173 # update Y
174 Y += mu * leq
175  
176 # update mu
177 mu = min(self.max_mu, mu * self.rho)
178  
179 return (E, alpha, niter, stop_c)

182 """Decompose a matrix M into low-rank, sparse and noise components using L1 factorization.
