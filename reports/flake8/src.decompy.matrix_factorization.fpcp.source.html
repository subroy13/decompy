31 def _shrink(self, v, lambdval):
32 """Shrinks values in v towards zero.
33  
34 This applies soft thresholding/shrinkage to each value in v.
35 Values that are less than the lambdval are set to zero.
36  
37 Parameters
38 ----------
39 v : numpy array
40 Array to apply shrinkage to.
41 lambdval : float
42 Shrinkage parameter. Values in v less than this are set to zero.
43  
44 Returns
45 -------
46 u : numpy array
47 Shrunk version of v.
48 """
49 u = np.sign(v) * np.maximum(0, np.abs(v) - lambdval)
