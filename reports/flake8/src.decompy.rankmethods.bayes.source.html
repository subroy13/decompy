10 def _gibbs_sample(E, Nu, Nv, phi, mu, psi, lor, llb=50, lub=750):
11 """
12 One step of gibbs sampling (restricted integral approximation)
13 E: (m, n), Nu: (m, m-r+1), Nv: (n, n-r+1)
14 phi, mu, psi: scalar
15 lor = log odds ratio
16 """

18 def lcr(theta, L):
19 n = theta.shape[0]
20 lc = np.zeros(L + 1)
21 ls = np.zeros(L + 1)
22 lr = np.zeros(L + 1)
23 lr[0] = 0
24 lc[0] = 0
25 for i in range(1, L + 1):
26 tmp = np.sum((theta / theta[0]) ** i)
27 ls[i] = i * np.log(theta[0]) + np.log(tmp)
28 tmp = np.sum(np.exp(lc + ls[::-1] - ls - np.log(2 * i)))
29 lc[i] = ls[i] + np.log(tmp)
30 lr[i] = lc[i] + loggamma(i + 1) + loggamma(n / 2) - loggamma(n / 2 + i)
31 return lr

33 m, n = E.shape
34  
35 E0 = Nu.T @ E @ Nv # (m-r+1, m) x (m, n) x (n, n-r+1) -> (m-r+1, n-r+1)
36 m0, n0 = E0.shape # m0 = m-r+1, n0 = n-r+1
37 svdE0 = np.linalg.svd(E0, full_matrices=False) # (m0, k), (k, ), (k, n0)
38 z = svdE0[1] ** 2 # (k, )
39 E02 = np.sum(z) # scalar
40 theta = z / E02 # (k, )

42 delta = 0
43 u = np.zeros(m0) # (m0, )
44 v = np.zeros(n0) # (n0, )
45  
46 # critical value for approximating bessel function
47 a = (svdE0[1] ** 2) * (phi**2) * np.max(z) # (k, )
48 lcrit = 0.5 * (
49 np.sqrt((n0 / 2 + 1) ** 2 + 4 * (a - n0 / 2)) - (n0 / 2 + 1)
50 ) # (k, )
51 lmax = 1.5 * np.max(lcrit)
52 lmax = int(min(max(lmax, llb), lub))
53 lseq = np.arange(lmax + 1) # (lmax + 1, )

55 # log terms
56 la = (
57 loggamma(m0 / 2)
58 - loggamma(m0 / 2 + lseq)
59 - loggamma(lseq + 1)
60 - 2 * lseq * np.log(2)
61 + lcr(theta, lmax)
62 ) # (lmax + 1, )
63 lb = (
64 ln2moment(mu * psi / (phi + psi), 1 / (phi + psi), lmax)
65 + 0.5 * np.log(psi / (phi + psi))
66 + 0.5 * mu**2 * psi * phi / (phi + psi)
67 ) # (lmax + 1, )
68 lc = lseq * np.log(E02 * phi**2) # (lmax + 1, )

70 lper10 = np.max(lt) + np.log(np.sum(np.exp(lt - np.max(lt)))) # scalar

77 pdl = np.exp(lpd - np.max(lpd)) # (lmax + 1, )

91 pvals /= pvals.sum()

93 np.arange(svdE0D.shape[0]), p=pvals
94 ) # random index between 1 to k
95 v = (svdE0[2][j, :] * (-1 if np.random.random() < 0.5 else 1)).reshape(
96 -1
97 ) # (n0, )
98 for _ in range(25):
99 u = rmf_vector(A @ v) # (m0, n0) x (n0, ) -> (m0, )
100 v = rmf_vector(A.T @ u) # (n0, m0) x (m0, ) -> (n0, )

104 def _gibbs_fixedrank(U, VT, D, Y, phi, mu, psi):
105 """
106 ## Gibbs Sample Fixed Rank
107 Y: (m, n), U: (m, r), D: (r, ), VT: (r, n)
108 phi, mu, psi: scalar
109 """
110 m, n = Y.shape
111 indices = np.arange(D.shape[0])[D != 0] # (r, )
112 for j in indices:
113 if len(indices) == 1:
114 Nu = np.identity(m) # r = 1, so (m, m)
115 Nv = np.identity(n) # r = 1, so (n, n)
116 else:
117 Nu = null_space(np.delete(U, j, 1).T) # Null((r-1, m)) -> (m, m-r+1)
118 Nv = null_space(np.delete(VT, j, 0)) # Null((r-1, n)) -> (n, n-r+1)
119  
120 E = Y - np.delete(U, j, 1) @ np.diag(np.delete(D, j, 0)) @ np.delete(
121 VT, j, 0
122 ) # (m, n) - (m, r-1) x (r-1, r-1) x (r-1, n)
123 VT[j, :] = Nv @ rmf_vector(
124 D[j] * phi * (Nv.T @ E.T @ U[:, j])
125 ) # (n, n-r+1) x rmf( (n-r+1, n) x (n, m) x (m, ) )
126 U[:, j] = Nu @ rmf_vector(
127 D[j] * phi * (Nu.T @ E @ VT[j, :])
128 ) # (m, m-r+1) x rmf( (m-r+1, m) x (m, n) x (n, ) )
129  
130 mn = (U[:, j].reshape(1, -1) @ E @ VT[j, :].reshape(-1, 1) * phi + mu * psi) / (
131 phi + psi
132 ) # (1, m) x (m, n) x (n, 1)
133 se = np.sqrt(1 / (phi + psi))
134 d = np.random.randn() * se + mn
135 D[j] = d
136 return (U, VT, D)

139 def _gibbs_varrank(U, VT, D, Y, phi, mu, psi, nsamp):
140 """
141 # Gibbs Variable Rank
142 Y: (m, n), U: (m, r), D: (r, ), VT: (r, n)
143 phi, mu, psi: scalar
144 nsamp = effective value of r
145 """
146 m, n = Y.shape
147 indices = np.random.choice(np.arange(n), size=nsamp) # (nsamp, )
148 for j in indices:
149 if len(indices) == 1:
150 Nu = np.identity(m)
151 Nv = np.identity(n)
152 else:
153 Nu = null_space(np.delete(U, j, 1).T) # Null((r-1, m)) -> (m, m-r+1)
154 Nv = null_space(np.delete(VT, j, 0)) # Null((r-1, n)) -> (n, n-r+1)
155  
156 E = Y - np.delete(U, j, 1) @ np.diag(np.delete(D, j, 0)) @ np.delete(VT, j, 0)
157  
158 # log odds ratio
159 nonzero_d = int(np.sum(D != 0) - (D[j] != 0))
160 lor = np.log((nonzero_d + 1) / (n - nonzero_d))
161 D[j], tmpu, tmpv = _gibbs_sample(
162 E, Nu, Nv, phi, mu, psi, lor
163 ) # scalar, (m-r+1, ), (n-r+1,)
164 U[:, j] = Nu @ tmpu # (m, m-r+1) x (m-r+1) -> (m, )
165 VT[j, :] = Nv @ tmpv # (n, n-r+1) x (n-r+1) -> (n, )
166 return (U, VT, D)

169 def rank_hoffbayes(Y, svdfunc, gibbstype="fixed", kmax=20, verbose=False):
170 """
171 Rank estimation using Bayesian method
172 - Reference: https://www.jstor.org/stable/27639896
173 """
174 assert gibbstype in ["fixed", "var"], "Invalid gibbs type"
175  
176 # set up the hyperparameters
177 sY = svdfunc(Y)
178 m, n = Y.shape
179 kmax = min(m, n, kmax)
180 s20est, t20est, mu0est = np.zeros(n), np.zeros(n), np.zeros(n)
181 s20est[0], t20est[0], mu0est[0] = np.var(Y), 0, 0
182 for k in range(1, n):
183 s20est[k] = np.var(Y - sY[0][:, :k] @ np.diag(sY[1][:k]) @ sY[2][:k, :])
184 t20est[k] = np.var(sY[1][:k])
185 mu0est[k] = np.mean(sY[1][:k])
186  
187 nu0, s20 = 2, np.mean(s20est) # prior for phi
188 eta0, t20 = 2, np.mean(t20est) # prior for psi
189 mu0, premu0 = np.mean(mu0est), 1 / np.var(mu0est) # prior for mu
190  
191 # MCMC parameters
192 nburnin = 100
193 nsamp = 100
194 nthin = 10
195 dvals = np.zeros((nsamp, kmax))
196 niter = nburnin + (nsamp + 1) * nthin
197  
198 # set up the initial values
199 mu = mu0
200 phi = 1 / s20
201 psi = 1 / t20
202  
203 if gibbstype == "fixed":
204 U = sY[0][:, :kmax]
205 D = sY[1][:kmax]
206 VT = sY[2][:kmax, :]
207 elif gibbstype == "var":
208 U = np.zeros((m, n))
209 VT = np.zeros((n, n))
210 D = np.zeros(n)
211 U[:, :kmax] = sY[0][:, :kmax]
212 D[:kmax] = sY[1][:kmax]
213 VT[:kmax, :] = sY[2][:kmax, :]
214  
215 ncount = 0
216 iterrange = tqdm(range(niter)) if verbose else range(niter)
217 for it in iterrange:
218 if gibbstype == "var":
219 U, VT, D = _gibbs_varrank(U, VT, D, Y, phi, mu, psi, kmax)
220 U, VT, D = _gibbs_fixedrank(U, VT, D, Y, phi, mu, psi)
221 if it > nburnin and it % nthin == 0:
222 dvals[ncount, :] = D
223 ncount += 1
224  
225 currank = np.sum(D != 0)
226 phi = np.random.gamma(
227 (nu0 + m * n) / 2,
228 scale=2 / (nu0 * s20 + np.sum((Y - U @ np.diag(D) @ VT) ** 2)),
229 ) # update phi
230 mu = np.random.randn() / np.sqrt(premu0 + psi * currank) + (
231 premu0 * mu0 + psi * np.sum(D)
232 ) / (
233 premu0 + psi * currank
234 ) # update mu
235 psi = np.random.gamma(
236 (eta0 + currank) / 2, 2 / (eta0 * t20 + np.sum((D[D != 0]) ** 2))
237 ) # update psi
238  
239 # finally look at the zeros
240 tol = 1e-5
241 if gibbstype == "fixed":
242 quants = np.quantile(dvals, q=[0.025, 0.975], axis=0)
243 rank = np.sum((quants[0, :] < 0) & (quants[1, :] > 0))
244 elif gibbstype == "var":
245 rank = mode(np.sum((np.abs(dvals) <= tol), axis=1), axis=None)
246 return kmax - rank
